30void FastNonAccelStepper::begin(uint8_t stepPin, uint8_t dirPin, bool invertMotorDir) {
31
32 // set dir logic for counting
33 // 1) If invertMotorDir == false, the DIR_PIN == HIGH, when motor moving forward and DIR_PIN == LOW, when motor moving backwards
34 // 2) If invertMotorDir == false, the PCNT counter should go up, when DIR_PIN == HIGH
35 if (false == invertMotorDir)
36 {
37 _dir_level_forward_b = LOW;
38 _dir_level_backward_b = HIGH;
39 _dir_pcnt_lctrl_mode_b = PCNT_MODE_KEEP;
40 _dir_pcnt_hctrl_mode_b = PCNT_MODE_REVERSE;
41 }
42 else
43 {
44 _dir_level_forward_b = HIGH;
45 _dir_level_backward_b = LOW;
46 _dir_pcnt_lctrl_mode_b = PCNT_MODE_REVERSE;
47 _dir_pcnt_hctrl_mode_b = PCNT_MODE_KEEP;
48 }
49
50 // init MCPWM and PCNTs
51 initMCPWM();
52 initPCNTMultiturn();
53 initPCNTControl();
54
55 pinMode(_stepPin, OUTPUT);
56 pinMode(_dirPin, OUTPUT);
57 digitalWrite(_stepPin, LOW);
58 digitalWrite(_dirPin, LOW);
59
60 // configure pin modes
61 mcpwm_gpio_init(MCPWM_UNIT_0, MCPWM0A, _stepPin);
62
63 // connect PCNT to GPIO pin
64 gpio_iomux_in(_stepPin, PCNT_SIG_CH0_IN0_IDX);
65 gpio_iomux_in(_stepPin, PCNT_SIG_CH0_IN1_IDX);
66
67 // reset pcnt counters
68 pcnt_counter_clear(PCNT_UNIT_0);
69 pcnt_counter_clear(PCNT_UNIT_1);
70
71 // make sure mcpwm is stopped
72 forceStop();
73
74}

90void FastNonAccelStepper::move(long stepsToMove, bool blocking) {
91
92 // stop previous move
93 forceStop();
94
95 long absStepsToMove = abs(stepsToMove);
96
97 if (absStepsToMove > POSITION_TRIGGER_THRESHOLD) {
98
99 // calculate number of required pcnt wraps
100 long numbWraps = 0; //absStepsToMove / PCNT_MIN_MAX_THRESHOLD;
101
102 long absStepsToMove_helper = absStepsToMove;
103 for (uint8_t idx = 0; idx < 100; idx++)
104 {
105 if (absStepsToMove_helper > PCNT_MIN_MAX_THRESHOLD)
106 {
107 absStepsToMove_helper -= PCNT_MIN_MAX_THRESHOLD;
108 numbWraps++;
109 }
110 else
111 {
112 break;
113 }
114 }
115
116 long limit;
117 if (numbWraps > 0)
118 {
119 limit = absStepsToMove / numbWraps;
120 }
121 else
122 {
123 limit = absStepsToMove;
124 }
125
126 int16_t limit_i16 = constrain(limit, 0, PCNT_MIN_MAX_THRESHOLD);
127
128
129
130 int16_t highLimit;
131 int16_t lowLimit;
132 // 1) set DIR pin
133 // 2) define upper limit for control pcnt
134 // 3) define lower limit for control pcnt
135 if (stepsToMove > 0) {
136 digitalWrite(_dirPin, _dir_level_forward_b);
137 highLimit = limit_i16;//absPositionChange - 1;
138 lowLimit = -MCPWM_PCNT_MAX_ALLOWED_MOVEMENT_IN_OPPOSITE_DIR_TILL_STOP;
139 _overflowCountControl = numbWraps;
140 } else {
141 digitalWrite(_dirPin, _dir_level_backward_b);
142 highLimit = MCPWM_PCNT_MAX_ALLOWED_MOVEMENT_IN_OPPOSITE_DIR_TILL_STOP;
143 lowLimit = -limit_i16;//-(absPositionChange - 1);
144 _overflowCountControl = numbWraps;
145 }
146
147 // parameterize control pcnt
148 pcnt_counter_pause(PCNT_UNIT_1);
149 pcnt_counter_clear(PCNT_UNIT_1);
150 pcnt_set_event_value(PCNT_UNIT_1, PCNT_EVT_H_LIM, highLimit);
151 pcnt_set_event_value(PCNT_UNIT_1, PCNT_EVT_L_LIM, lowLimit);
152 pcnt_event_enable(PCNT_UNIT_1, PCNT_EVT_H_LIM);
153 pcnt_event_enable(PCNT_UNIT_1, PCNT_EVT_L_LIM);
154 pcnt_counter_clear(PCNT_UNIT_1);
155 pcnt_counter_resume(PCNT_UNIT_1);
156
157 //Serial.printf("Hlim: %d, LLim: %d, wraps: %d\n", highLimit, lowLimit, numbWraps);
158
159 // start mcpwm
160 _isRunning = true;
161 mcpwm_start(MCPWM_UNIT_0, MCPWM_TIMER_0);
162
163 if (blocking)
164 {
165 while(isRunning())
166 {
167 delay(1);
168
169 /*int16_t pulseCountlcl = 0;
170 pcnt_get_counter_value(PCNT_UNIT_1, &pulseCountlcl);
171 Serial.printf( "CurPos: %d, CtrlPos:%d, overfl: %d\n", getCurrentPosition(), pulseCountlcl, _overflowCountControl);
172 delay(30);*/
173 }
174 }
