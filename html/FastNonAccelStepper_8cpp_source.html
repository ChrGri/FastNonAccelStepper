29void FastNonAccelStepper::begin(uint8_t stepPin_u8, uint8_t dirPin_u8, bool invertMotorDir_b)
30{
31 // set dir logic for counting
32 // 1) If invertMotorDir_b == false, the DIR_PIN == HIGH, when motor moving forward and DIR_PIN == LOW, when motor moving backwards
33 // 2) If invertMotorDir_b == false, the PCNT counter should go up, when DIR_PIN == HIGH
34 if (false == invertMotorDir_b)
35 {
36 dirLevelForward_b = LOW;
37 dirLevelBackward_b = HIGH;
38 dirPcntLctrlMode_u8 = PCNT_MODE_KEEP;
39 dirPcntHctrlMode_u8 = PCNT_MODE_REVERSE;
40 }
41 else
42 {
43 dirLevelForward_b = HIGH;
44 dirLevelBackward_b = LOW;
45 dirPcntLctrlMode_u8 = PCNT_MODE_REVERSE;
46 dirPcntHctrlMode_u8 = PCNT_MODE_KEEP;
47 }
48
49 // init MCPWM and PCNTs
50 initMCPWM();
51 initPCNTMultiturn(); // unit to track the overall position
52 initPCNTControl(); // unit to controll the step bursts
53
54 pinMode(stepPin_u8, OUTPUT);
55 pinMode(dirPin_u8, OUTPUT);
56 digitalWrite(stepPin_u8, LOW);
57 digitalWrite(dirPin_u8, LOW);
58
59 // configure pin modes
60 mcpwm_gpio_init(MCPWM_UNIT_0, MCPWM0A, stepPin_u8);
61
62 // connect PCNT to GPIO pin
63 gpio_iomux_in(stepPin_u8, PCNT_SIG_CH0_IN0_IDX); // overall position unit
64 gpio_iomux_in(stepPin_u8, PCNT_SIG_CH0_IN1_IDX); // controll unit
65
66 // reset pcnt counters
67 pcnt_counter_clear(PCNT_UNIT_0); // overall position unit
68 pcnt_counter_clear(PCNT_UNIT_1); // controll unit
69
70 // make sure mcpwm is stopped
71 forceStop();
72}

99{
100 // stop previous move
101 forceStop();
102
103 int32_t absStepsToMove_i32 = abs(stepsToMove_i32);
104
105 if (absStepsToMove_i32 > POSITION_TRIGGER_THRESHOLD)
106 {
107 // MATLAB code:
108 // PCNT_MIN_MAX_THRESHOLD = 32767; absStepsToMove = 2*PCNT_MIN_MAX_THRESHOLD; nmbWraps = idivide( int32(absStepsToMove) , int32(PCNT_MIN_MAX_THRESHOLD) ); limit = absStepsToMove / (nmbWraps+1)
109 // close all; clear all; clc;
110 // PCNT_MIN_MAX_THRESHOLD = 32767;
111 // absStepsToMove = int32( (0:1:1000) * PCNT_MIN_MAX_THRESHOLD );
112 // nmbWraps = idivide( int32(absStepsToMove), int32(PCNT_MIN_MAX_THRESHOLD) );
113 // limit = absStepsToMove ./ (nmbWraps+1);
114 // nexttile()
115 // plot(limit)
116 // nexttile()
117 // plot(absStepsToMove, (nmbWraps+1) .* limit )
118 // title('Step output')
119 // xlabel('True steps \rightarrow')
120 // ylabel('Actual steps \rightarrow')
121 // nexttile()
122 // plot( (nmbWraps+1) .* limit - absStepsToMove)
123 // title('Step Error')
124
125 // compute the number of wraps
126 int32_t numbWraps_i32 = absStepsToMove_i32 / PCNT_MIN_MAX_THRESHOLD;
127
128 // divide into equally spaced bursts, e.g. when
129 // if absStepsToMove is < PCNT_MIN_MAX_THRESHOLD --> numbWraps = 0 and limit_i16 = absStepsToMove
130 // if absStepsToMove == PCNT_MIN_MAX_THRESHOLD --> numbWraps = 1 and limit_i16 = absStepsToMove/2
131 // if absStepsToMove == 2*PCNT_MIN_MAX_THRESHOLD --> numbWraps = 2 and limit_i16 = absStepsToMove/3
132 // if absStepsToMove > PCNT_MIN_MAX_THRESHOLD --> numbWraps >= 1
133 int32_t limit_i32 = absStepsToMove_i32 / (numbWraps_i32 + 1);
134 int16_t limit_i16 = constrain(limit_i32, 0, PCNT_MIN_MAX_THRESHOLD);
135
136 // the highLimit | lowLimit will be hit numbWraps, before stopping the PWM output
137 int16_t highLimit_i16;
138 int16_t lowLimit_i16;
139
140 // 1) set DIR pin
141 // 2) define upper limit for control pcnt
142 // 3) define lower limit for control pcnt
143 if (stepsToMove_i32 > 0)
144 {
145 digitalWrite(dirPin_u8, dirLevelForward_b);
146 highLimit_i16 = limit_i16;
147 lowLimit_i16 = -MCPWM_PCNT_MAX_ALLOWED_MOVEMENT_IN_OPPOSITE_DIR_TILL_STOP;
148 overflowCountControl_i32 = numbWraps_i32;
149 }
150 else
151 {
152 digitalWrite(dirPin_u8, dirLevelBackward_b);
153 highLimit_i16 = MCPWM_PCNT_MAX_ALLOWED_MOVEMENT_IN_OPPOSITE_DIR_TILL_STOP;
154 lowLimit_i16 = -limit_i16;
155 overflowCountControl_i32 = numbWraps_i32;
156 }
157
158 // parameterize control pcnt
159 pcnt_counter_pause(PCNT_UNIT_1);
160 pcnt_counter_clear(PCNT_UNIT_1);
161 pcnt_set_event_value(PCNT_UNIT_1, PCNT_EVT_H_LIM, highLimit_i16);
162 pcnt_set_event_value(PCNT_UNIT_1, PCNT_EVT_L_LIM, lowLimit_i16);
163 pcnt_event_enable(PCNT_UNIT_1, PCNT_EVT_H_LIM);
164 pcnt_event_enable(PCNT_UNIT_1, PCNT_EVT_L_LIM);
165 pcnt_counter_clear(PCNT_UNIT_1);
166 pcnt_counter_resume(PCNT_UNIT_1);
167

169
170 // start mcpwm
171 delayMicroseconds(5);
172 isRunning_b = true;
173 mcpwm_start(MCPWM_UNIT_0, MCPWM_TIMER_0);
174
175 if (blocking_b)
176 {
